<objective>
Build the Configure step UI for validation rules: a disclosure-style ValidationRuleEditor on each field row, a batch-level corrector toggle + cap input next to the existing PromptTemplateEditor, Zustand state extension and persistence, and template save/load round-tripping of `field_rules`.

Purpose: This is the curator-facing entry point for the entire validation feature. Without this UI, no rules can be attached and the engine built in Plan 02 has nothing to operate on. After this ships, curators can configure rules, save them with templates, and ship batches that emit per-field validation outcomes.
Output: A field row in Configure step expands to reveal a rule editor with preset picker, custom regex, vocabulary list, fuzzy toggle, and a per-field corrector opt-in. The Configure step's "How to extract" card gains a batch-level corrector toggle + cap input. Templates round-trip rules cleanly.
</objective>

       acceptCorrectorProposal: (filename, field) => set((state) => ({
         results: state.results.map(r => {
           if (r.filename !== filename) return r;
           const proposal = r.validation?.[field]?.corrector_proposal;
           const newEdited = { ...r.editedData };
           if (proposal != null) newEdited[field] = proposal;
           const newValidation = r.validation ? { ...r.validation } : null;
           if (newValidation && newValidation[field]) {
             newValidation[field] = { ...newValidation[field], status: 'valid' };
           }
           return { ...r, editedData: newEdited, validation: newValidation };
         }),
       })),
       rejectCorrectorProposal: (filename, field) => set((state) => ({
         results: state.results.map(r => {
           if (r.filename !== filename) return r;
           const newValidation = r.validation ? { ...r.validation } : null;
           if (newValidation && newValidation[field]) {
             newValidation[field] = { ...newValidation[field], status: 'invalid' };
           }
           return { ...r, validation: newValidation };
         }),
       })),

      { id: 'viaf_id',     label: 'VIAF ID',                pattern: '^\\d+$',                    description: 'VIAF numeric identifier' },
      { id: 'prefix',      label: 'Prefix Pattern',         pattern: '',                          description: 'User-supplied prefix + digits', hasPrefixInput: true },

      { id: 'vocabulary',  label: 'Closed Vocabulary',      pattern: '',                          description: 'Match against allowed values' },

    export function buildPrefixPattern(prefix: string): string {
      // Escape regex special chars in user-supplied prefix
      const esc = prefix.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');

    }
