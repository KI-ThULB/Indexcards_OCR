    CREATE `apps/frontend/src/features/configure/AuthorityBindingEditor.tsx`:

    This component follows the exact same pattern as `ValidationRuleEditor.tsx` — a collapsible disclosure section per field row. Read ValidationRuleEditor.tsx to match its collapse toggle pattern (`isExpanded` state + ChevronDown icon + border-t divider).

    ```tsx
    import { useState } from 'react';
    import { ChevronDown, ChevronRight, Link2 } from 'lucide-react';
    import { AuthorityBinding, AuthorityType } from '../../api/batchesApi';
    import { MetadataField } from '../../store/wizardStore';

    interface AuthorityBindingEditorProps {
      field: MetadataField;
      onChange: (binding: AuthorityBinding | null) => void;
    }

    // 9 options: None + 5 GND sub-collections + 3 external authorities
    const AUTHORITY_OPTIONS: Array<{ value: AuthorityType; label: string }> = [
      { value: null,                   label: 'None' },
      { value: 'gnd-persons',          label: 'GND: Persons' },
      { value: 'gnd-places',           label: 'GND: Places' },
      { value: 'gnd-subjects',         label: 'GND: Subjects' },
      { value: 'gnd-corporate-bodies', label: 'GND: Corporate Bodies' },
      { value: 'gnd-works',            label: 'GND: Works' },
      { value: 'wikidata',             label: 'Wikidata' },
      { value: 'geonames',             label: 'GeoNames' },
      { value: 'aat',                  label: 'Getty AAT' },
    ];

    export function AuthorityBindingEditor({ field, onChange }: AuthorityBindingEditorProps) {
      const [isExpanded, setIsExpanded] = useState(false);
      const currentType = field.authority?.type ?? null;
      const currentLabel = AUTHORITY_OPTIONS.find(o => o.value === currentType)?.label ?? 'None';

      return (
        <div className="border-t border-parchment-dark/10">
          {/* Collapse header */}
          <button
            type="button"
            onClick={() => setIsExpanded(p => !p)}
            className="flex items-center justify-between w-full px-6 py-2.5 text-left hover:bg-parchment-light/40 transition-colors"
          >
            <div className="flex items-center gap-1.5 text-xs text-archive-600">
              <Link2 size={12} className="text-archive-400" />
              <span>Authority</span>
              {currentType && (
                <span className="ml-1 px-1.5 py-0.5 rounded text-xs bg-blue-100 text-blue-700 font-medium">
                  {currentLabel}
                </span>
              )}
            </div>
            {isExpanded
              ? <ChevronDown size={14} className="text-archive-400" />
              : <ChevronRight size={14} className="text-archive-400" />
            }
          </button>

          {/* Expanded content */}
          {isExpanded && (
            <div className="px-6 pb-3 pt-1">
              <label className="block text-xs text-archive-600 mb-1">Authority file</label>
              <select
                value={currentType ?? ''}
                onChange={(e) => {
                  const val = e.target.value || null;
                  onChange(val ? { type: val as AuthorityType } : null);
                }}
                className="w-full border border-archive-300 rounded px-2 py-1.5 text-sm bg-parchment-paper text-archive-800 focus:outline-none focus:ring-1 focus:ring-archive-500"
              >
                {AUTHORITY_OPTIONS.map(opt => (
                  <option key={opt.value ?? '_none'} value={opt.value ?? ''}>
                    {opt.label}
                  </option>
                ))}
              </select>
              <p className="mt-1 text-xs text-archive-500">
                When set, the Clean step can reconcile this field against the authority file and emit URIs in exports.
              </p>
            </div>
          )}
        </div>
      );
    }
    ```
